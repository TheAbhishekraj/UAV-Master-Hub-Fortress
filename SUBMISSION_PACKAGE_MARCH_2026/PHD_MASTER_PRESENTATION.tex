\begin{frame}{Navigation: A* Path Planning in Bihar Fields}
    The drone navigates rows by minimizing:
    \begin{center}
        \huge $f(n) = g(n) + h(n)$
    \end{center}
    \begin{itemize}
        \item $g(n)$: Distance from landing pad.
        \item $h(n)$: Euclidean heuristic to thermal maize hotspot.
        \item Obstacle avoidance validated against tractor-polygon models.
    \end{itemize}
\end{frame}

% Slide 6: Mathematical Foundation
\begin{frame}{Mathematical Foundation: ES-EKF on $SO(3)$}
    Implementation of Error-State Kalman Filter for GPS-denied states:
    \begin{block}{State Vector Update}
        $$\delta \mathbf{x} = [\delta \mathbf{p}, \delta \mathbf{v}, \delta \boldsymbol{\theta}, \delta \mathbf{a}_b, \delta \boldsymbol{\omega}_b]^\top$$
        Rotation update using Lie Groups:
        $$\mathbf{R}_{next} = \mathbf{R} \exp([\delta \boldsymbol{\theta}]_\times)$$
    \end{block}
    \textit{Ensures global orientation consistency without gimbal lock singularities.}
\end{frame}
