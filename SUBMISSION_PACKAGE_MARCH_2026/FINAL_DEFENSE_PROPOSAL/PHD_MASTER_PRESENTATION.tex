\begin{frame}{Slide 6: Path Planning (A* Navigation)}
    \begin{itemize}
        \item Dynamic obstacle avoidance (e.g., tractors, trees).
        \item We implemented an $A^*$ search where the cost function is optimized for maize row adherence:
    \end{itemize}
    \begin{equation}
        f(n) = g(n) + h(n)
    \end{equation}
    \begin{itemize}
        \item \textit{$g(n)$}: Exact cost from start.
        \item \textit{$h(n)$}: Heuristic cost to target (Euclidean bias).
    \end{itemize}
\end{frame}

\begin{frame}{Slide 7: The 'Inner Ear' - ES-EKF on SO(3)}
    \textbf{The Challenge:} Standard EKFs suffer gimbal lock in high winds.
    
    \textbf{The Solution:} Error-State Kalman Filter on the $SO(3)$ manifold.
    \begin{equation}
        \delta \mathbf{x} = [\delta \mathbf{p}, \delta \mathbf{v}, \delta \boldsymbol{\theta}, \delta \mathbf{a}_b, \delta \boldsymbol{\omega}_b]^\top
    \end{equation}
    \begin{itemize}
        \item Maintains attitude stability singularity-free.
        \item Z-Variance validated at \textbf{0.031m} (< 0.05m target).
    \end{itemize}
\end{frame}

\begin{frame}{Slide 9: Resolving QoS Starvation}
    \begin{itemize}
        \item \textbf{Bug:} "IMU Updates: 0" causing total Navigation failure.
        \item \textbf{Root Cause:} FastRTPS Volatile QoS mismatch dropping IMU packets before node subscription.
        \item \textbf{Fix:} Engineered \texttt{TRANSIENT\_LOCAL} policy ensuring persistence of crucial sensor streams.
    \end{itemize}
\end{frame}
